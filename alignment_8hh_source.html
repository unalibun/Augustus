149 struct SortCriterion {
150  SortCriterion(size_t speciesIdx) : s(speciesIdx) {};
151  bool operator() (Alignment* const a1, Alignment* const a2){
152  // alignments, where row s is missing come last
153  if (a2->rows[s] == NULL)
154  return true;
155  if (a1->rows[s] == NULL)
156  return false;
157  if (a1->rows[s]->seqID < a2->rows[s]->seqID)
158  return true;
159  if (a1->rows[s]->seqID > a2->rows[s]->seqID)
160  return false;
161  // same sequence, compare chromosomal start positions
162  return (a1->rows[s]->chrStart() < a2->rows[s]->chrStart());
163  }
164  size_t s;
165 };

170 struct IdxSortCriterion {
171  IdxSortCriterion(vector<Alignment*> const a_, size_t speciesIdx) : s(speciesIdx), a(a_) {};
172  bool operator() (int i, int j){
173  // alignments, where row s is missing come last
174  if (a[j]->rows[s] == NULL && a[i]->rows[s]) // this conjunction makes sorting stable where the sequence is missing
175  return true;
176  if (a[i]->rows[s] == NULL)
177  return false;
178  if (a[i]->rows[s]->seqID < a[j]->rows[s]->seqID)
179  return true;
180  if (a[i]->rows[s]->seqID > a[j]->rows[s]->seqID)
181  return false;
182  // same sequence, compare chromosomal start positions
183  return (a[i]->rows[s]->chrStart() < a[j]->rows[s]->chrStart());
184  }
185  size_t s;
186  vector<Alignment*> const &a;
